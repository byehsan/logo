Install

npm (GitHub Packages)

# .npmrc
@byehsan:registry=https://npm.pkg.github.com

# install
npm install @byehsan/logo

ESM import

import {
  base, stateLoading,
  http404, palettes
} from '@byehsan/logo'

Create lockups

import {
  createTextLockup,
  createIconLockup
} from '@byehsan/logo'

const blog = createTextLockup('blog')
const git  = createIconLockup(
  'git', gitIconSvgString)

CSS custom properties

import '@byehsan/logo/css'

/* then in CSS */
.logo {
  background: var(--logo-base);
  color: var(--color-orange);
}

Favicon HTML

<link rel="icon"
      type="image/svg+xml"
      href="base.svg">
<link rel="apple-touch-icon"
      href="apple-touch-icon.png">
<link rel="manifest"
      href="site.webmanifest">

Theming via currentColor

<!-- Set color on SVG or parent -->
<img src="lockup-git.svg"
     style="color:#f07828">

<!-- Or drive from CSS -->
.brand { color: var(--color-orange); }
