Install

npm (GitHub Packages)

# .npmrc
@byehsan:registry=https://npm.pkg.github.com

# install
npm install @byehsan/logo

Sub-brand lockup generator

import { subbrandLockup } from '@byehsan/logo'

// transparent SVG string
const svg = subbrandLockup('blog')

// with options
const svg2 = subbrandLockup('store', {
  palette: 'brand-light',
  bgColor: '#eeeef4',
  fontSize: 20,
})

import {

  http404, palettes
} from '@byehsan/logo'
